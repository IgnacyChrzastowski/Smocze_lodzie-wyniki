if ($_SERVER['REQUEST_METHOD'] === 'POST'
    && isset($_POST['action'])
    && $_POST['action'] === 'add_zawody') {

    $nazwa = trim($_POST['nazwa_zawodow'] ?? '');

    if ($nazwa === '') {
        $alert = "Podaj nazwę zawodów.";
    } else {

        $conn->query("UPDATE zawody SET status='zakonczone'");

        $stmt = $conn->prepare("
            INSERT INTO zawody (nazwa, status)
            VALUES (?, 'aktywne')
        ");

        $stmt->bind_param("s", $nazwa);

        if ($stmt->execute()) {

            $new_id = $conn->insert_id;

            $stmt2 = $conn->prepare("
                INSERT INTO ustawienia (klucz, wartosc)
                VALUES ('aktywne_zawody', ?)
                ON DUPLICATE KEY UPDATE wartosc=VALUES(wartosc)
            ");

            $wartosc = (string)$new_id;
            $stmt2->bind_param("s", $wartosc);
            $stmt2->execute();

            header("Location: ".$_SERVER['PHP_SELF']);
            exit;
        }
    }
}

ZMIANY W index.php:
dodać:
const selectZawody = document.getElementById('zawodySelect');

function loadZawodyList() {

    fetch('ajax_get_zawody.php')
        .then(r => r.json())
        .then(data => {

            selectZawody.innerHTML = '';

            data.forEach(z => {

                const option = document.createElement('option');

                option.value = z.id;

                option.textContent =
                    z.nazwa +
                    (z.status === 'aktywne'
                        ? ' (AKTYWNE)'
                        : ' (zakończone)');

                if (parseInt(z.id) === parseInt(currentZawodyId)) {
                    option.selected = true;
                }

                selectZawody.appendChild(option);
            });
        });
}
